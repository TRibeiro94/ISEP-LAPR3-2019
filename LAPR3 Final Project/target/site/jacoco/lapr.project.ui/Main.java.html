package lapr.project.ui;

import java.sql.SQLException;
import java.util.ArrayList;
import lapr.project.controller.RentalController;
import lapr.project.controller.VehicleController;
import lapr.project.data.RentalDataHandler;
import lapr.project.data.VehicleDataHandler;
import lapr.project.model.GeographicalLocation;
import lapr.project.model.PointInterest;

/**
 *
 * @author TiagoRibeiro
 */
public class Main {
    
    public static void main(String[] args) throws ClassNotFoundException, SQLException {
        RentalController rc = new RentalController(new RentalDataHandler());
        VehicleController vc = new VehicleController(new VehicleDataHandler());
        PointInterest p1 = new PointInterest(381, new GeographicalLocation(41.15227, -8.60929, 104), "Parque da Trindade");
        PointInterest p2 = new PointInterest(388, new GeographicalLocation(41.29527, -8.69372, 15), "Outlet Vila do Conde");
        ArrayList<PointInterest> pois = new ArrayList<>();
        PointInterest po = new PointInterest(387, new GeographicalLocation(41.13646, -8.61199, 0), "Cais de Gaia");
        PointInterest p3= new PointInterest(407, new GeographicalLocation(41.14571, -8.61496, 25), "Torre dos Clerigos");
        PointInterest p4 = new PointInterest(408, new GeographicalLocation(41.1687, -8.68622, 0), "Edificio Transparente");
        PointInterest p5 = new PointInterest(384, new GeographicalLocation(41.16837, -8.59375, 9), "ISEP");
        PointInterest pd = new PointInterest(388, new GeographicalLocation(41.29527, -8.69372, 15), "Outlet Vila do Conde");
        pois.add(p3);
        pois.add(p4);
        pois.add(p5);
        System.out.println(rc.shortestRoute(p1, p2, 2));
        System.out.println("---------------------------------");
        System.out.println(rc.shortestRoutePassingThroughPoints(po, pd, pois));
        System.out.println("---------------------------------");
        System.out.println(rc.mostEnergeticallyEfficientRoute(8, vc.getScooterByDescription("ePT0010"), po, p3));
        System.out.println("---------------------------------");
        System.out.println(rc.mostEfficientRoutePassingThroughPoints(75, vc.getScooterByDescription("ePT0010"), po, pd, pois, 3, true, "energy"));
        LoginUI login = new LoginUI();
        login.loop();
    }
}
